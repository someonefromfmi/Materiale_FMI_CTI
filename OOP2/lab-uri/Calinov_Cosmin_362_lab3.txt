// Unboxing
import java.util.ArrayList;
import java.util.List;

public class Unboxing {

    public static void main(String[] args) {
        Integer i = new Integer(-8);

        // 1. Unboxing through method invocation
        int absVal = absoluteValue(i);
        System.out.println("absolute value of " + i + " = " + absVal);

        List<Double> ld = new ArrayList<>();
        ld.add(3.1416);    // Π is autoboxed through method invocation.

        // 2. Unboxing through assignment
        double pi = ld.get(0);
        System.out.println("pi = " + pi);
    }

    public static int absoluteValue(int i) {
        return (i < 0) ? -i : i;
    }
}

4. package org.models;

import java.time.Year;
import java.util.ArrayList;
import java.util.List;

abstract class Contract {
    private static int counter = 0;
    protected int nrContract;
    protected int anSemnare;

    public Contract(int anSemnare) {
        this.nrContract = ++counter;
        this.anSemnare = anSemnare;
    }

    public abstract boolean esteValid();
    
    @Override
    public String toString() {
        return "Contract #" + nrContract + ", An: " + anSemnare;
    }
}

class ContractAngajareNedeterminata extends Contract {
    private String angajat;

    public ContractAngajareNedeterminata(int anSemnare, String angajat) {
        super(anSemnare);
        this.angajat = angajat;
    }

    @Override
    public boolean esteValid() {
        return true; // Contractele nedeterminate sunt mereu valide
    }
}

class ContractAngajareDeterminata extends Contract {
    private String angajat;
    private int luniValabilitate;

    public ContractAngajareDeterminata(int anSemnare, String angajat, int luniValabilitate) {
        super(anSemnare);
        this.angajat = angajat;
        this.luniValabilitate = luniValabilitate;
    }

    @Override
    public boolean esteValid() {
        int anExpirare = anSemnare + (luniValabilitate / 12);
        return anExpirare >= Year.now().getValue();
    }
}

class ContractCumparare extends Contract {
    private String produs;

    public ContractCumparare(int anSemnare, String produs) {
        super(anSemnare);
        this.produs = produs;
    }

    @Override
    public boolean esteValid() {
        return true; // Contractele de cumpărare sunt definitive
    }
}

class ContractInchiriere extends Contract {
    private String locatie;
    private int luniValabilitate;

    public ContractInchiriere(int anSemnare, String locatie, int luniValabilitate) {
        super(anSemnare);
        this.locatie = locatie;
        this.luniValabilitate = luniValabilitate;
    }

    @Override
    public boolean esteValid() {
        int anExpirare = anSemnare + (luniValabilitate / 12);
        return anExpirare >= Year.now().getValue();
    }
}

class ContractManager {
    private List<Contract> contracte = new ArrayList<>();

    public void adaugaContract(Contract contract) {
        contracte.add(contract);
    }

    public void afiseazaContracte() {
        for (Contract contract : contracte) {
            System.out.println(contract);
        }
    }

    public int numarContracte() {
        return contracte.size();
    }

    public int numarContracteValide() {
        return (int) contracte.stream().filter(Contract::esteValid).count();
    }

    public int numarContracteInvalide() {
        return numarContracte() - numarContracteValide();
    }
}

public class Main {
    public static void main(String[] args) {
        ContractManager manager = new ContractManager();
        
        manager.adaugaContract(new ContractAngajareNedeterminata(2022, "Ion Popescu"));
        manager.adaugaContract(new ContractAngajareDeterminata(2020, "Maria Ionescu", 24));
        manager.adaugaContract(new ContractCumparare(2021, "Laptop Dell"));
        manager.adaugaContract(new ContractInchiriere(2019, "Apartament Bucuresti", 36));
        
        System.out.println("Lista contracte:");
        manager.afiseazaContracte();

        System.out.println("Total contracte: " + manager.numarContracte());
        System.out.println("Contracte valide: " + manager.numarContracteValide());
        System.out.println("Contracte invalide: " + manager.numarContracteInvalide());
    }
}

package org.example;

class A {
    int x;

    public A(int x) {
        this.x = x;
    }
};

class B {
    int x;

    public B(int x) {
        this.x = x;
    }
};

class C implements Cloneable {
    A a;
    B b;

    public C(B b, A a) {
        this.b = b;
        this.a = a;
    }

    @Override
    public String toString() {
        return "C{" +
                "a=" + a +
                ", b=" + b +
                '}';
    }

    @Override
    protected Object clone() throws CloneNotSupportedException {
        return new  C(new B(this.b.x), new A(this.a.x));
    }
};

public class Main {

    public static void main(String[] args) {
        A a = new A(10);
        B b = new B(10);

        C c = new C(b, a);
        System.out.println(c);

        try {
            System.out.println(c.clone());
        } catch (CloneNotSupportedException e) {
            throw new RuntimeException(e);
        }
    }
}
